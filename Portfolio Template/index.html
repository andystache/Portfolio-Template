FizzBuzz

Enter two numbers between 1 and 100 in the boxes below. Press "Go" to see multiples of the first number revealed as "Fizz" and multiples of the second number revealed as "buzz". Multiples of both numbers will be revealed as FizzBuzz.

Your calculated results


    $("#FBcode").hide();
    $("#FBbtnToggleCode").click(function () {
    $("#FBcode").toggle();

    if ($(this).text() == "Show Code") {
    $(this).text("Hide Code");

    }
    else {
    $(this).text("Show Code")
    };
    });
    $("#btnFB").click(function () {
    let Input1 = Number($("#fbIn1").val());
    let Input2 = Number($("#fbIn2").val());
    let Answer = [];
    for (var i = 1; i <= 100; i++) {
    var int1 = i % Input1;
    var int2 = i % Input2;
    if (int1 == 0 && int2 == 0) {
    Answer.push('FizzBuzz');
    }
    else if (int1 == 0) {
    Answer.push('Fizz');
    }
    else if (int2 == 0) {
    Answer.push('Buzz');
    }
    else {
    Answer.push(`${i}`);
    };
    };
    $("#Answer").html(Answer.join(', '));
    });
    $("#FBbtnClear").click(function () {
    $("#fbIn1").val("");
    $("#fbIn2").val("");
    $("#Answer").text("");
    });

Factorial!

To calculate a factorial, enter a number and press "calculate."

Your calculated results


    $("#factCode").hide();
    $("#facttoggleBtn").click(function () {
    $("#factCode").toggle();
    if ($(this).text() == "Show Code") {
    $(this).text("Hide Code");
    }
    else {
    $(this).text("Show Code")
    };
    });
    $("#btnFact").click(function () {
    //Step 1: Get the user data
    var number = Number($("#fact").val());

    //I am using an If statement to evaluate something
    if (number > 170) {
    alert("Please enter positive numbers less than 171!");
    $("#fact").val("");
    return;
    }

    //Step 2: Perform the algorithm
    for (var foo = number - 1; foo > 1; foo--) {
    //How do we calculate the factorial
    number *= foo;
    }


    //Step 3: Show the result
    $("#factOut").text(number);
    //Clear button
    });
    $("#factbtnClear").click(function () {
    $("#fact").val("");
    $("#factOut").text("");
    });

Please enter a word to determine if it is a palindrome.

    //Palindrome
    $("#palcode").hide();
    $("#paltoggleBtn").click(function () {
    $("#palcode").toggle();
    if ($(this).text() == "Show Code") {
    $(this).text("Hide Code");
    }
    else {
    $(this).text("Show Code")
    };
    });
    $("#palIn").click(function () {
    let inputWord = $("#pal").val()
    if (inputWord.length < 2) { alert("Please use two or more characters") } else {
    let reversedInput = inputWord.split('').reverse().join('');
    let word = inputWord.toLowerCase().replace(/\s|\'|\"/g, '');
    let revword = reversedInput.toLowerCase().replace(/\s|\'|\"/g, '');
    let message = `Your input was: ${inputWord}. Reversed is ${reversedInput}.`;
    let result = ""
    if (word == revword) {
    result = `That means ${inputWord} is a palindrome!`;
    } else {
    result = `That means ${inputWord} is not a plaindrome!`;
    };
    $("#palOut").text(message + " " + result)
    }
    });
    $("#palbtnClear").click(function () {
    $("#pal").val("");
    $("#palOut").text("");
    });
