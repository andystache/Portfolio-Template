        if ($(this).text() == "Show Code") {
            $(this).text("Hide Code");

        }
        else {
            $(this).text("Show Code")
        };
    });

    $("#btnFB").click(function () {
        //1. Get user data
        var Input1 = Number($("#fbIn1").val());
        var Input2 = Number($("#fbIn2").val());
        var Answer = "";


        //2. Do something with the input

        for (var i = 1; i <= 100; i++) {
            var comma = ", ";
            if (i == 100) {
                comma = "";
            }
            if (i % Input1 == 0 && i % Input2 == 0) {
                Answer += "fizzbuzz" + comma + "";
            }

            else if (i % Input1 == 0) {
                Answer += "fizz" + comma + "";

            }
            else if (i % Input2 == 0) {
                Answer += "buzz" + comma + "";
                    
            }

            else {
                Answer += i + comma;
            }
        }
     //3. Show the answer.
    $("#Answer").html(Answer);   
        

    });

    //Clear out the answers to start over
    $("#FBbtnClear").click(function () {
        $("#fbIn1").val("");
        $("#fbIn2").val("");
        $("#Answer").text("");
    });
    //Fizz Buzz ends

    //Step 1: Get the user data

    if (number > 170) {
    alert("Please enter positive numbers less than 171!");
    $("#fact").val("");

    }

    //Step 2: Perform the algorithm
    for (var foo = number - 1; foo > 1; foo--) {

$("#palIn").click(function () {
    let inputWord = $("#pal").val()
        if (inputWord.length < 2) { 
            alert("Please use two or more characters") } 
        else {
            let reversedInput = inputWord.split('').reverse().join('');
            let word = inputWord.toLowerCase().replace(/\s/g, '');
            let revword = reversedInput.toLowerCase().replace(/\s/g, '');
            let message = `Your input was: ${inputWord}. Reversed is ${reversedInput}.`;
            let result = ""
            if (word == revword) {
                result = `... ${inputWord} is a palindrome!`;
            } else {
                result = `... ${inputWord} is not a plaindrome!`;
            };
        $("#palOut").text(message + " " + result)
        }
});

    $("#pal").val("");
    $("#palOut").text("");
});
